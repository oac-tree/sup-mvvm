# Quick start

```C++

class GaussianDistribution 
{
public:
  GaussianDistribution() : xc(0.0), yc(0.0), radius(1.0) {}
  Circle(double xc, double yc, double radius) : xc(xc_), yc(yc_), radius(radius) {}

  double SetMean(double mean) const {mean_ = radius}
  double GetRadius() const {return radius_}

private:
  double mean_;
  double std_dev_;
}
```
